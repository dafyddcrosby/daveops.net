===== Setting RAM on Vagrant =====
'''
driver:
  name: vagrant
  customize:
    memory: 2048
'''


===== Setting environment in chef-zero =====

'''
driver:
  name: vagrant

provisioner:
  name: chef_zero
  environments_path: path/to/environments
  client_rb:
    environment: production
'''
